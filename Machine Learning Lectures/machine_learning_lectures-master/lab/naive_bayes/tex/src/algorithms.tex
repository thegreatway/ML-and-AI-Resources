\begin{algorithm}[H]
\begin{algorithmic}[1]
\FOR{$i=1$ to $n$}
\FOR{$j=1$ to $d$}
\STATE compute $\log{P(X_i, Y_j)}$
\STATE compute $\log{P(Y_j/X_i)} = \log{P(X_i, Y_j)} + \log{P(Y_j)}$
\ENDFOR
\STATE $Y_i=\argmax_{Y_j}\log{P(Y_j/X_i)}$
\ENDFOR
\end{algorithmic}
\caption{pseudocode for inference}
\label{alg:test}
\end{algorithm}
