\begin{frame}{Use logarithms!}
All those products of probabilities can result in nothing. Use log-probabilities instead!
\begin{equation*}
\log{P(X_i/Y_c)}=\sum_{j=1}^m \log{P(x_i^{(j)}/Y_c)}
\end{equation*}
\begin{equation*}
\log{P(Y/X)} = \log{P(X/Y)} + \log{P(Y)}
\end{equation*}
\end{frame}
