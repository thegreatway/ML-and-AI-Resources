\begin{frame}{Adaboost: a closer look (2)}
To compute the error and the efficiency of a weak classifier $M_j$
\begin{equation*}
\epsilon_j = \sum_{M_j(x_i)\neq y_i} w_i
\end{equation*}
\begin{equation*}
\alpha_j = \frac{1}{2} \ln\bigg(\frac{1-\epsilon_j}{\epsilon_j}\bigg)
\end{equation*}
\end{frame}
